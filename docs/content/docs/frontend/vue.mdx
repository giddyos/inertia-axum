The CLI scaffold is a CSR-focused `src/main.ts` application with an explicit resolver. These steps show the maintained production-example shape: `@inertiajs/vite`, a custom `src/app.js` entry, client and SSR builds, and browser hydration coverage.

## Configure Vue

<Steps>
  <Step>
    ### Install the tested packages

    ```bash title="Terminal"
    pnpm --dir frontend add @inertiajs/vue3@3.6.1 vue@3.5.29
    pnpm --dir frontend add -D @inertiajs/vite@3.6.1 @vitejs/plugin-vue@6.0.5 vite@8.1.4
    ```
  </Step>
  <Step>
    ### Configure Vite

    ```js title="frontend/vite.config.js"
    import inertia from '@inertiajs/vite'
    import vue from '@vitejs/plugin-vue'
    import { defineConfig } from 'vite'

    export default defineConfig({
      plugins: [vue(), inertia()],
      build: {
        manifest: true,
        rollupOptions: {
          input: 'src/app.js',
        },
      },
    })
    ```
  </Step>
  <Step>
    ### Boot Inertia

    ```js title="frontend/src/app.js"
    import { createInertiaApp } from '@inertiajs/vue3'

    createInertiaApp()
    ```
  </Step>
  <Step>
    ### Render the shared Home page

    <Snippet language="vue" source="vue/src/Pages/Home.vue" title="frontend/src/Pages/Home.vue" />

    The matching server contract is in the [quick start](/docs/getting-started/quick-start).
  </Step>
  <Step>
    ### Build

    ```bash title="Terminal"
    pnpm --dir frontend build
    ```
  </Step>
  <Step>
    ### Continue to production rendering

    Compare [`axum-vue`](https://github.com/giddyos/inertia-axum/tree/main/examples/axum-vue) and add its client/SSR entries through the [SSR frontend build](/docs/ssr/frontend-build).
  </Step>
</Steps>
