The CLI scaffold is a CSR-focused `src/main.ts` application with an explicit page resolver. These steps show the maintained production-example shape: `@inertiajs/vite`, a custom `src/app.js` entry, client and SSR build support, and browser hydration coverage.

## Configure Svelte

<Steps>
  <Step>
    ### Install the tested packages

    ```bash title="Terminal"
    pnpm --dir frontend add @inertiajs/svelte@3.6.1
    pnpm --dir frontend add -D @inertiajs/vite@3.6.1 @sveltejs/vite-plugin-svelte@7.1.2 svelte@5.55.7 svelte-preprocess@6.0.5 vite@8.1.4
    ```
  </Step>
  <Step>
    ### Configure Vite

    ```js title="frontend/vite.config.js"
    import inertia from '@inertiajs/vite'
    import { svelte } from '@sveltejs/vite-plugin-svelte'
    import { defineConfig } from 'vite'

    export default defineConfig({
      plugins: [svelte({ prebundleSvelteLibraries: true }), inertia()],
      build: {
        manifest: true,
        rolldownOptions: {
          input: 'src/app.js',
        },
      },
    })
    ```
  </Step>
  <Step>
    ### Boot Inertia

    ```js title="frontend/src/app.js"
    import { createInertiaApp } from '@inertiajs/svelte'

    const pages = import.meta.glob('./Pages/**/*.svelte', { eager: true })
    createInertiaApp({ resolve: (name) => pages[`./Pages/${name}.svelte`] })
    ```
  </Step>
  <Step>
    ### Render the shared Home page

    <Snippet language="svelte" source="svelte/src/Pages/Home.svelte" title="frontend/src/Pages/Home.svelte" />

    The matching server contract is in the [quick start](/docs/getting-started/quick-start).
  </Step>
  <Step>
    ### Build

    ```bash title="Terminal"
    pnpm --dir frontend build
    ```
  </Step>
  <Step>
    ### Continue to production rendering

    Compare the [`axum-svelte` source](https://github.com/giddyos/inertia-axum/tree/main/examples/axum-svelte) and add its client/SSR entries through the [SSR frontend build](/docs/ssr/frontend-build).
  </Step>
</Steps>
